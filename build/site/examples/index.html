On Off Switch

View Source 
<!--
This is our DOM element
In this case it's just a `button` that is bound to the `on-off-switch` component

We bind the `click` event to the `toggle` method of the controller
-->
<button is="on-off-switch" @click="this.toggle">LOADING...</button>

<script type="module">
    import { registerControllers, makeController } from '/static/js/binder/binder.js';

    class OnOffSwitch extends makeController(HTMLButtonElement, "button") {
        // The `init` method is called when the controller is first created
        // Use this to handle your setup logic
        init() {
            this.isOn = true;
            this.root.innerText = "ON";
        }

        // This is the click handler for the button
        toggle() {
            this.isOn = !this.isOn;
            this.root.innerText = this.isOn ? "ON" : "OFF";
        }
    }

    registerControllers(OnOffSwitch);
</script>

Click Counter

Clicks: ${this.data.clicks}
View Source 
<!-- HTML -->
<click-counter>
    <p @render>Clicks: ${this.data.clicks}</p>
    <button @click.prevent="this.count()">Click me!</button>
</click-counter>

<!-- JavaScript -->
<script type="module">
    import { registerControllers, Controller } from '/static/js/binder/binder.js';

    class ClickCounter extends Controller {
        init() {
            this.data.clicks = 0;
        }

        count() {
            this.data.clicks++;
            this.render();
        }
    }

    registerControllers(ClickCounter);
</script>

<!-- HTML -->
<currency-picker :currency="GBP">
    <select @bind.render="this.data.currency">
        <option disabled selected>Please select a currency</option>
        <option value="AUD">Australian Dollars</option>
        <option selected value="GBP">British Pounds</option>
        <option value="EUR">Euros</option>
        <option value="USD">US Dollars</option>
    </select>

    <h4 @render>Chosen currency: ${this.data.currency}</h4>
</currency-picker>

<!-- JavaScript -->
<script type="module">
    // This example uses the core Controller component
    import { registerControllers, Controller } from '/static/js/binder/binder.js';

    class CurrencyPicker extends Controller {
        init() {
            this.data.currency = this.args.currency;
        }
    }

    registerControllers(CurrencyPicker);
</script>

Magic 8 Ball

${this.randomAnswer()}
View Source 
<magic-8-ball>
    <button @render @click="this.render">${this.prompt}</button>
    <p @render.eval>${this.randomAnswer()}</p>
</magic-8-ball>

<script type="module">
    import { registerControllers, Controller } from '/static/js/binder/binder.js';

    class Magic8Ball extends Controller {
        init() {
            this.prompt = 'Ask the magic 8 ball';

            this.answers = [
                "It is certain.",
                "It is decidedly so.",
                "Without a doubt.",
                "Yes definitely.",
                "You may rely on",
                "As I see it, yes.",
                "Most likely.",
                "Outlook good.",
                "Yes.",

                "Signs point to yes.",
                "Reply hazy, try again.",
                "Ask again later.",
                "Better not tell you now.",
                "Cannot predict now.",
                "Concentrate and ask again.",

                "Don't count on it.",
                "My reply is no.",
                "My sources say no.",
                "Outlook not so good.",
                "Very doubtful.",
            ];
        }

        randomAnswer() {
            return this.answers[Math.floor(Math.random() * this.answers.length)]
        }
    }

    registerControllers(Magic8Ball);
</script>

Dynamic Frame

The line below is another page loaded within a div, click here to load the page directly!
View Source 
<!-- HTML -->
<dynamic-frame
    :url="/current_date"
    :mount-point="#mount-here"
    :auto-refresh="5s"
    :execute-scripts="true">

    <p>The line below is another page loaded within a div, click <a href='/current_date'>here</a> to load the page directly!</p>

    <div id="mount-here"></div>
</dynamic-frame>

<button id="refresh-dynamic-frame">Force Refresh</button>


<!-- JavaScript -->
<script type="module">
    import { registerControllers } from '/static/js/binder/binder.js';
    import { DynamicFrame } from '/static/js/binder/core/dynamic_frame.js';

    registerControllers(DynamicFrame);

    document.querySelector("#refresh-dynamic-frame").addEventListener("click", () => {
        document.querySelector("dynamic-frame").refresh();
    });
</script>
